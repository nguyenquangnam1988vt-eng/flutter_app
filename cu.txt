const MethodChannel _nativeChannel = MethodChannel('driver_monitor/native');

void main() async {
  WidgetsFlutterBinding.ensureInitialized();
  await _initNotifications();
  runApp(const MyApp());
}

Future<void> _initNotifications() async {
  const android = AndroidInitializationSettings('@mipmap/ic_launcher');
  const ios = DarwinInitializationSettings();
  const initSettings = InitializationSettings(android: android, iOS: ios);
  await FlutterLocalNotificationsPlugin().initialize(initSettings);
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Driver Phone Monitor',
      theme: ThemeData(
        primarySwatch: Colors.blue,
        brightness: Brightness.dark,
      ),
      home: const Dashboard(),
      debugShowCheckedModeBanner: false,
    );
  }
}

class Dashboard extends StatefulWidget {
  const Dashboard({super.key});

  @override
  State<Dashboard> createState() => _DashboardState();
}

class _DashboardState extends State<Dashboard> with WidgetsBindingObserver {
  double x = 0, y = 0, z = 0;
  double speed = 0;
  bool alert = false;
  bool monitoringBackground = false;
  bool _appInForeground = true;

  final List<double> _tiltHistory = [];
  static const int _tiltBufferSize = 50; // Gi·∫£m ƒë·ªÉ ph·∫£n h·ªìi nhanh h∆°n
  double _lastTilt = 1.0;

  final List<double> _yBuffer = [];
  static const int _bufferSize = 30;

  final List<double> _yBufferLong = [];
  static const int _bufferLongSize = 60;

  StreamSubscription<AccelerometerEvent>? _accelSub;
  StreamSubscription<Position>? _posSub;

  ScreenMonitor? _screenMonitor;

  final FlutterLocalNotificationsPlugin _localNotif =
      FlutterLocalNotificationsPlugin();
  final AudioPlayer _audioPlayer = AudioPlayer();

  @override
  void initState() {
    super.initState();
    _requestPermissions();
    WidgetsBinding.instance.addObserver(this);

    // ‚öôÔ∏è C·∫•u h√¨nh AudioPlayer tr∆∞·ªõc khi s·ª≠ d·ª•ng
    _audioPlayer
      ..setReleaseMode(ReleaseMode.stop)
      ..setVolume(1.0);

    if (Platform.isIOS) {
      _setupNativeCallback();
      _screenMonitor = ScreenMonitor();
      _screenMonitor!.start((on) {
        setState(() {
          // Gi·ªØ l·∫°i cho t∆∞∆°ng th√≠ch
        });
      });
    }

    _listenSensors();
    _listenLocation();
  }

  @override
  void didChangeAppLifecycleState(AppLifecycleState state) {
    setState(() {
      _appInForeground = state == AppLifecycleState.resumed;
    });
    debugPrint('App state: $state, In foreground: $_appInForeground');
  }

  Future<void> _requestPermissions() async {
    await Permission.locationAlways.request();
    await Permission.sensors.request();
    await Permission.notification.request();
  }

  void _setupNativeCallback() {
    _nativeChannel.setMethodCallHandler((call) async {
      if (call.method == 'backgroundSpeedAlert') {
        final args = call.arguments as Map<dynamic, dynamic>?;
        final speed = (args?['speed'] ?? 0).toDouble();
        _showLocalAlert(speed, 1.0);
      }
    });
  }

  Future<void> _showLocalAlert(double spd, double tilt) async {
    final androidDetails = AndroidNotificationDetails(
      'driver_alerts',
      'Driver Alerts',
      importance: Importance.max,
      priority: Priority.high,
      playSound: true,
    );
    final iosDetails = DarwinNotificationDetails(
      presentAlert: true,
      presentBadge: true,
      presentSound: true,
      sound: 'alert.mp3',
    );

    await _localNotif.show(
      999,
      '‚ö†Ô∏è C·∫£nh b√°o',
      'Ph√°t hi·ªán s·ª≠ d·ª•ng ƒëi·ªán tho·∫°i khi di chuy·ªÉn ·ªü ${spd.toStringAsFixed(1)} km/h (nghi√™ng ${(tilt * 100).toStringAsFixed(1)}%)',
      NotificationDetails(android: androidDetails, iOS: iosDetails),
    );

    await _audioPlayer.play(AssetSource('assets/alert.mp3'));

    if (await Vibration.hasVibrator() ?? false) {
      Vibration.vibrate(duration: 1000);
    }
  }

  void _listenSensors() {
    _accelSub = accelerometerEvents.listen((event) {
      setState(() {
        x = event.x;
        y = event.y;
        z = event.z;
      });

      // üéØ S·ª¨A L·ªñI: Ki·ªÉm tra gi√° tr·ªã c·∫£m bi·∫øn h·ª£p l·ªá
      double g = sqrt(x * x + y * y + z * z);
      
      // üö® QUAN TR·ªåNG: Ki·ªÉm tra tr√°nh chia cho 0
      double tilt = 1.0; // Gi√° tr·ªã m·∫∑c ƒë·ªãnh
      if (g > 0.1) { // Ch·ªâ t√≠nh tilt n·∫øu g ƒë·ªß l·ªõn
        tilt = (z / g).abs();
        // Gi·ªõi h·∫°n tilt trong kho·∫£ng 0-1
        tilt = tilt.clamp(0.0, 1.0);
      }

      debugPrint('SENSORS - x: ${x.toStringAsFixed(2)}, y: ${y.toStringAsFixed(2)}, z: ${z.toStringAsFixed(2)}, g: ${g.toStringAsFixed(2)}, tilt: ${(tilt * 100).toStringAsFixed(1)}%');

      _yBuffer.add(y);
      if (_yBuffer.length > _bufferSize) _yBuffer.removeAt(0);

      _yBufferLong.add(y);
      if (_yBufferLong.length > _bufferLongSize) _yBufferLong.removeAt(0);

      _checkAlert(g, tilt);
    }, onError: (error) {
      debugPrint('‚ùå Accelerometer error: $error');
    }, cancelOnError: false);
  }

  void _listenLocation() async {
    try {
      if (!await Geolocator.isLocationServiceEnabled()) {
        debugPrint('Location services disabled');
        return;
      }

      LocationPermission p = await Geolocator.checkPermission();
      if (p == LocationPermission.denied)
        p = await Geolocator.requestPermission();
      if (p == LocationPermission.denied || p == LocationPermission.deniedForever) {
        debugPrint('Location permission denied');
        return;
      }

      const settings = LocationSettings(
        accuracy: LocationAccuracy.bestForNavigation,
        distanceFilter: 5, // TƒÉng l√™n ƒë·ªÉ gi·∫£m battery
      );

      _posSub = Geolocator.getPositionStream(locationSettings: settings)
          .listen((Position pos) {
        double newSpeed = (pos.speed >= 0) ? pos.speed * 3.6 : 0.0;
        setState(() {
          speed = newSpeed;
        });
        debugPrint('Location - Speed: ${speed.toStringAsFixed(1)} km/h');
      }, onError: (error) {
        debugPrint('‚ùå Location error: $error');
      });

    } catch (e) {
      debugPrint('‚ùå Location setup error: $e');
    }
  }

  double _calculateStdDev(List<double> values) {
    if (values.isEmpty) return 0.0;
    double mean = values.reduce((a, b) => a + b) / values.length;
    double sumSquaredDiff =
        values.fold(0.0, (sum, val) => sum + pow(val - mean, 2));
    return sqrt(sumSquaredDiff / values.length);
  }

  void _checkAlert(double g, double tilt) async {
    // üéØ S·ª¨A L·ªñI: Ki·ªÉm tra buffer c√≥ d·ªØ li·ªáu kh√¥ng
    if (_tiltHistory.isEmpty && tilt > 0) {
      // Kh·ªüi t·∫°o buffer v·ªõi gi√° tr·ªã hi·ªán t·∫°i
      for (int i = 0; i < _tiltBufferSize; i++) {
        _tiltHistory.add(tilt);
      }
    }

    _tiltHistory.add(tilt);
    if (_tiltHistory.length > _tiltBufferSize) _tiltHistory.removeAt(0);

    // üéØ S·ª¨A L·ªñI: T√≠nh avgTilt an to√†n
    double avgTilt = _tiltHistory.isNotEmpty 
        ? _tiltHistory.reduce((a, b) => a + b) / _tiltHistory.length 
        : 1.0;

    bool overSpeed = speed > 5.0;
    bool gravityOK = g > 8.0 && g < 11.0;
    double stdDevYLong = _calculateStdDev(_yBufferLong);
    bool stdDevYStable = stdDevYLong <= 1.5;

    // ‚úÖ X√°c ƒë·ªãnh tr·∫°ng th√°i nghi√™ng
    bool isTilted = avgTilt < 0.5;

    debugPrint('ALERT CHECK - Speed: $overSpeed, Tilted: $isTilted, AvgTilt: ${(avgTilt * 100).toStringAsFixed(1)}%, GravityOK: $gravityOK, Stable: $stdDevYStable, AppActive: $_appInForeground');

    bool shouldAlert = overSpeed && isTilted && gravityOK && stdDevYStable && _appInForeground;

    if (shouldAlert) {
      if (!alert) {
        debugPrint('üö® ALERT TRIGGERED!');
        await _showLocalAlert(speed, avgTilt);
      }
      setState(() => alert = true);
    } else {
      if (alert) {
        debugPrint('‚úÖ ALERT CLEARED');
      }
      setState(() => alert = false);
    }
    
    _lastTilt = avgTilt;
  }

  Future<void> startBackgroundMonitoring() async {
    if (!Platform.isIOS) return;
    try {
      await _nativeChannel.invokeMethod('startBackground');
      setState(() => monitoringBackground = true);
      debugPrint('Background monitoring started');
    } on PlatformException catch (e) {
      debugPrint('startBackground failed: $e');
    }
  }

  Future<void> stopBackgroundMonitoring() async {
    if (!Platform.isIOS) return;
    try {
      await _nativeChannel.invokeMethod('stopBackground');
      setState(() => monitoringBackground = false);
      debugPrint('Background monitoring stopped');
    } on PlatformException catch (e) {
      debugPrint('stopBackground failed: $e');
    }
  }

  @override
  void dispose() {
    _accelSub?.cancel();
    _posSub?.cancel();
    _audioPlayer.dispose();
    WidgetsBinding.instance.removeObserver(this);
    if (Platform.isIOS) {
      _screenMonitor?.stop();
      _screenMonitor = null;
    }
    super.dispose();
  }

  Color getAlertColor() => alert ? Colors.redAccent : Colors.greenAccent;

  @override
  Widget build(BuildContext context) {
    double stdDevYShort = _calculateStdDev(_yBuffer);
    double stdDevYLong = _calculateStdDev(_yBufferLong);
    double g = sqrt(x * x + y * y + z * z);
    
    // üéØ S·ª¨A L·ªñI: T√≠nh tilt hi·ªÉn th·ªã an to√†n
    double displayTilt = 1.0;

      displayTilt = (z / g).abs().clamp(0.0, 1.0);
    }
    
    // T√≠nh avgTilt cho hi·ªÉn th·ªã

        ? _tiltHistory.reduce((a, b) => a + b) / _tiltHistory.length

      body: Padding(
        padding: const EdgeInsets.all(16.0),
        child: Column(

                color: getAlertColor(),

              ),
              alignment: Alignment.center,

                mainAxisAlignment: MainAxisAlignment.center,

                        : "‚úÖ An to√†n",
                    style: const TextStyle(
                      color: Colors.black,
                      fontWeight: FontWeight.bold,
                      fontSize: 18,
                    ),
                  ),
                  Text(
                    "Tilt: ${(avgTilt * 100).toStringAsFixed(1)}%",

              children: [
                Expanded(
                  child: ElevatedButton(
                    onPressed: monitoringBackground ? null : startBackgroundMonitoring,
                    child: const Text('B·∫Øt ƒë·∫ßu gi√°m s√°t n·ªÅn'),
                  ),
                ),
                const SizedBox(width: 10),
                Expanded(
                  child: ElevatedButton(
                    onPressed: monitoringBackground ? stopBackgroundMonitoring : null,
                    style: ElevatedButton.styleFrom(backgroundColor: Colors.redAccent),
                    child: const Text('D·ª´ng'),
                  ),
                ),
              ],
            ),
          ],
        ),
      ),
    );
  }
